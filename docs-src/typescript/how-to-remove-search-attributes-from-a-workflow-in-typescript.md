```typescript
import {upsertSearchAttributes} from "@temporalio/workflow";

async function yourWorkflow() {
  upsertSearchAttributes({CustomIntField: [1, 2, 3]});

  // ... later, to remove:
  upsertSearchAttributes({CustomIntField: []});
}
```
